select 
RAZAOSOCIALENTIDADE as NOMERAZAO ,
CNPJCPF AS CPFCNPJ,
IERG AS RGIE,
NOMEENTIDADE AS APELIDOFANTASIA,
CEP,
ENDERECO,
NUMERO,
'' AS COMPLEMENTO,
BAIRRO,
CODIGOCIDADE,
NOMECIDADE AS CIDADE,
UF,
FONE AS TELEFONE,
EMAIL,
OBS AS OBSERVACAO,
'' AS MENSAGEM,
CASE WHEN PGTO_DINHEIRO = 1 THEN 'S' ELSE 'N' END AS CLIENTEVISTA,
CASE WHEN PGTO_PRAZO = 1 THEN 'S' ELSE 'N' END AS CLIENTEPRAZO,
CASE WHEN CLIENTE_FORNECEDOR = 1 OR CLIENTE_FORNECEDOR = 2 THEN 'S' ELSE 'N' END AS FORNECEDOR,
CASE WHEN PGTO_CHQ_PRE = 1 OR PGTO_CHQ_VISTA = 1 THEN 'S' ELSE 'N' END AS CLIENTECHEQUE,
LIMITE AS LIMITEONLINE,
LIMITE AS LIMITEOFFLINE
from entidades, CIDADES
WHERE ENTIDADES.ID_CIDADES = CIDADES.ID_CIDADES

select nome as NOMERAZAO
,CGC AS CPFCNPJ
,TRIM(CASE WHEN (IE <> 'ISENTO') THEN IE ELSE '' END) AS RGIE
,CASE WHEN (NOMEFANTASIA <> '') THEN NOMEFANTASIA ELSE NOME END AS APELIDOFANTASIA
,CEP
,Endereço AS ENDERECO
,Numero
,COMPLEMENTO
,BAIRRO
,C002_Municipios.C002_GeoCodigo AS IBGE
,CIDADE
,UF
,Telefone1 AS TELEFONE
,EMail
,OBS AS OBSERVACAO
,'' AS MENSAGEM
,'N' as CLIENTEVISTA
,'N' AS CLIENTEPRAZO
,'S' AS FORNECEDOR
,'N' AS CLIENTECHEQUE
,'0' AS LIMITEONLINE
,'0' AS LIMITEOFFLINE

FROM Fornecedores
LEFT OUTER JOIN C002_Municipios ON FORNECEDORES.UF = C002_MUNICIPIOS.C002_UF AND Fornecedores.CIDADE = C002_Municipios.C002_Nome

UNION

select nome as NOMERAZAO
,CASE WHEN (CGC <> '') THEN CGC ELSE CPF END AS CPFCNPJ
,TRIM(CASE WHEN (IE <> 'ISENTO') THEN IE ELSE '' END) AS RGIE
,CASE WHEN (NOMEFANTASIA <> '') THEN NOMEFANTASIA ELSE NOME END AS APELIDOFANTASIA
,CEP
,Endereço AS ENDERECO
,Numero
,COMPLEMENTO
,BAIRRO
,C002_Municipios.C002_GeoCodigo AS IBGE
,CIDADE
,UF
,Telefone1 AS TELEFONE
,EMail
,OBS AS OBSERVACAO
,'' AS MENSAGEM
,'S' as CLIENTEVISTA
,CASE WHEN ((LIMITE > 0) OR (Requisição = 1)) AND BLOQUEADO = 0 THEN 'S' ELSE 'N' END AS CLIENTEPRAZO
,'N' AS FORNECEDOR
,CASE WHEN CHEQUE = 1 and BLOQUEADO = 0 THEN 'S' ELSE 'N' END AS CLIENTECHEQUE
,CASE WHEN BLOQUEADO = 1 THEN '0.01' ELSE Limite END AS LIMITEONLINE
,CASE WHEN BLOQUEADO = 1 THEN '0.01' ELSE Limite END AS LIMITEOFFLINE
FROM Clientes
LEFT OUTER JOIN C002_Municipios ON CLIENTES.UF = C002_MUNICIPIOS.C002_UF AND CLIENTES.CIDADE = C002_Municipios.C002_Nome

ORDER BY 3 DESC, 2

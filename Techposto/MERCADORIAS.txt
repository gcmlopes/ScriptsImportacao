select 
CASE WHEN PRODUTOS.CODIGODEBARRA <> '' THEN CodigoDeBarra ELSE produtos.Código END as CODIGOBARRA
,''as CODIGORAPIDO
,produtos.Descrição as DESCRICAOCOMPLETA
,produtos.Descrição AS DESCRICAORESUMIDA
,SUBGRUPOS.DESCRIÇÃO AS SUBGRUPO
,PRODUTOS.Valor_Venda AS PRECOVENDA
,'PREÇO À VISTA' AS TIPOPRECOVENDA
,PRODUTOS.Valor_Custo AS PRECOCUSTO
,PRODUTOS.UNIDADE AS UNIDADEVENDA
,PRODUTOS.NCM
,PRODUTOS.IDANP AS ANP
,PRODUTOS.CST_Saida AS CSTICMS
,PRODUTOS.Alíquota AS ICMS
,'' AS CSTPISCOFINS
,'' AS NATUREZARECEITAPISCOFINS
,CASE WHEN PRODUTOS.CST_Saida = '60' THEN '405' WHEN PRODUTOS.CST_SAIDA = '00' THEN '102' END AS CFOPVENDAPDV --ADICIONAR NOVOS CONFORME A BASE
,CASE WHEN PRODUTOS.CST_Saida = '60' THEN '405' WHEN PRODUTOS.CST_SAIDA = '00' THEN '102' END AS CFOPVENDARETAGUARDA --ADICIONAR NOVOS CONFORME A BASE
,CASE WHEN PRODUTOS.CST_Saida = '60' THEN '403' WHEN PRODUTOS.CST_SAIDA = '00' THEN '102' END AS CFOPPADRAOCOMPRA --ADICIONAR NOVOS CONFORME A BASE
from produtos
left outer join SubGrupos on produtos.SubGrupo = SubGrupos.Código
WHERE PRODUTOS.IsCombustivel = 0 AND 
PRODUTOS.IsSomenteDeUsoInterno = 0 --NÃO IMPORTAR CONSUMO INTERNO
